# query1 og query2 gir samme resultat: 1 rad
# query 3 gir et tomt resultat

# ikke noen join, henter en rad fra hver tabell, bruker geometriene inn i ST_Within-funksjonen
query1 = """
SELECT b.byggnr, g.gkretsnavn, ST_Within(b.wkb_geometry, g.wkb_geometry) AS ligger_innenfor
FROM bygning750 b, grunnkrets g
WHERE g.gkretsnavn = 'MAIHAUGEN' 
  AND b.byggnr = '7447868';
"""

# tradisjonell database-join (men tabellene har ingen felles datafelter, så join har ikke så stor hensikt), bruker geometriene inn i ST_Within-funksjonen
query2 = """
SELECT b.byggnr, g.gkretsnavn, ST_Within(b.wkb_geometry, g.wkb_geometry) AS ligger_innenfor
FROM bygning750 b
JOIN grunnkrets g
ON g.gkretsnavn = 'MAIHAUGEN'
WHERE b.byggnr = '7447868';
"""

# spatial join, eventuelle rader i resultatet er kun de bygningene som ligger innenfor Maihaugen grunnkrets. Vi trenger derfor ikke ST_Within i SELECT-delen.
query3 = """
SELECT b.byggnr, g.gkretsnavn
FROM grunnkrets AS g
INNER JOIN bygning750 AS b
ON ST_Within(b.wkb_geometry, g.wkb_geometry)
WHERE g.gkretsnavn = 'MAIHAUGEN'
  AND b.byggnr = '7447868';
"""

with engine.begin() as conn:
  df = pd.read_sql_query(sql=query2, con=conn)
df
