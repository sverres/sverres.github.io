function lamp_basis(x, y, lamp; h)
    lx, ly = lamp
    r2 = (x - lx)^2 + (y - ly)^2 + h^2
    return h / r2^(1.5)
end;

# --- Column vector for one lamp ---
function lamp_column(lamp, grid_points; h=4.0)
    [lamp_basis(x, y, lamp; h=h) for (x, y) in grid_points]
end;
